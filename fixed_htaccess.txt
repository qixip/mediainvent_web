# MediaInvent.ro - Next.js Static Export with _next Directory Access
# CRITICAL: Allow access to _next directory (blocked by default in many hosts)
<Directory "_next">
    Options Indexes FollowSymLinks
    AllowOverride All
    Require all granted
</Directory>

# Alternative method to allow _next access
RewriteEngine On

# Allow access to _next directory and all subdirectories
RewriteRule ^_next/ - [L]

# Force correct MIME types for CSS, JS and Fonts
AddType text/css .css
AddType application/javascript .js
AddType application/json .json
AddType font/opentype .otf
AddType font/truetype .ttf
AddType font/woff .woff
AddType font/woff2 .woff2

# Enable compression for better performance
<IfModule mod_deflate.c>
    AddOutputFilterByType DEFLATE text/plain
    AddOutputFilterByType DEFLATE text/html
    AddOutputFilterByType DEFLATE text/xml
    AddOutputFilterByType DEFLATE text/css
    AddOutputFilterByType DEFLATE application/xml
    AddOutputFilterByType DEFLATE application/xhtml+xml
    AddOutputFilterByType DEFLATE application/rss+xml
    AddOutputFilterByType DEFLATE application/javascript
    AddOutputFilterByType DEFLATE application/x-javascript
    AddOutputFilterByType DEFLATE application/json
</IfModule>

# Browser caching for static assets
<IfModule mod_expires.c>
    ExpiresActive on
    ExpiresByType text/css "access plus 1 year"
    ExpiresByType application/javascript "access plus 1 year"
    ExpiresByType image/png "access plus 1 year"
    ExpiresByType image/jpg "access plus 1 year"
    ExpiresByType image/jpeg "access plus 1 year"
    ExpiresByType image/gif "access plus 1 year"
    ExpiresByType image/svg+xml "access plus 1 year"
    ExpiresByType image/webp "access plus 1 year"
    ExpiresByType font/opentype "access plus 1 year"
    ExpiresByType font/truetype "access plus 1 year"
    ExpiresByType font/woff "access plus 1 year"
    ExpiresByType font/woff2 "access plus 1 year"
</IfModule>

# Security headers
<IfModule mod_headers.c>
    Header always set X-Content-Type-Options nosniff
    Header always set X-Frame-Options DENY
    Header always set X-XSS-Protection "1; mode=block"
    
    # Ensure CSS files are served with correct content type
    <FilesMatch "\.(css)$">
        Header set Content-Type "text/css"
    </FilesMatch>
    
    # Ensure JS files are served with correct content type
    <FilesMatch "\.(js)$">
        Header set Content-Type "application/javascript"
    </FilesMatch>
    
    # Ensure font files are served with correct content type
    <FilesMatch "\.(otf|ttf|woff|woff2)$">
        Header set Content-Type "font/opentype"
    </FilesMatch>
</IfModule>

# Prevent access to sensitive files but allow _next
<Files ".htaccess">
    Order allow,deny
    Deny from all
</Files>

# Error pages
ErrorDocument 404 /404.html

# Additional protection while allowing _next
<FilesMatch "^(\.htaccess|\.htpasswd|\.env)">
    Order allow,deny
    Deny from all
</FilesMatch>
